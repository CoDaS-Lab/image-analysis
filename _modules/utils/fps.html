        """
        DESCRIPTION:
            update timer and number of frames
        """
        if self.elapsed() >= 1:
            print('fps {}'.format(self.fps()))
            self._nframes = 0
            self._start = self._end
        else:
            self._nframes += 1

        """
        DESCRIPTION:
            calculate elapsed
        """
        self._end = time()
        return (self._end - self._start)
